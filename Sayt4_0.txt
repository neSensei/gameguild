const menuToggle = document.getElementById('menu-toggle');
const mainNav = document.getElementById('main-nav');

menuToggle.addEventListener('click', () => {
  mainNav.classList.toggle('show');
});

// Навигация по секциям
document.querySelectorAll('nav a').forEach(link => {
  link.addEventListener('click', (e) => {
    e.preventDefault();
    const sectionId = link.dataset.section;
    showSection(sectionId);
    // Закрыть меню на мобильных
    if (mainNav.classList.contains('show')) {
      mainNav.classList.remove('show');
    }
  });
});

function showSection(id) {
  document.querySelectorAll('.section').forEach(sec => {
    sec.classList.remove('active');
  });
  document.getElementById(id).classList.add('active');
}

// Пример динамического наполнения новостей, обзоров, гайдов
const newsContent = document.getElementById('news-content');
const reviewsContent = document.getElementById('reviews-content');
const guidesContent = document.getElementById('guides-content');

const mockArticles = [
  { title: 'Объявление о новой игре', content: 'Вышел новый трейлер...', category: 'news' },
  { title: 'Обзор: Новая RPG', content: 'Детальный анализ...', category: 'reviews' },
  { title: 'Гайд по выживанию', content: 'Как пройти трудные уровни...', category: 'guides' },
];

function populateArticles() {
  mockArticles.forEach(article => {
    const card = document.createElement('div');
    card.className = 'article-card';
    card.innerHTML = `
      <h3>${article.title}</h3>
      <p>${article.content}</p>
      <button onclick="alert('Подробнее о ${article.title}')">Подробнее</button>
    `;
    if (article.category === 'news') {
      newsContent.appendChild(card);
    } else if (article.category === 'reviews') {
      reviewsContent.appendChild(card);
    } else if (article.category === 'guides') {
      guidesContent.appendChild(card);
    }
  });
}

// Загрузка данных при запуске
window.addEventListener('load', () => {
  populateArticles();
});
